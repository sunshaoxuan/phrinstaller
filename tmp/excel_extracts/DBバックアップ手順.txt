		key=Value ※下記キーが修正しないでください※						
		DBUSER=postgres						→　データベースユーザー
		DBPASSWORD=passwordxxx						→　データベースパスワード
		DBNAME_OHR=ohr						→　データベース名（ohr）
		DBNAME_TENANT=tenant						→　データベース名（tenant）
		DBSERVER=localhost						→　データベースIP
		PORT=5432  						→　データベースポート
		POSTGRE_PATH=E:\Program Files\PostgreSQL\16\bin						→　postgresインストールフォルダのbinパスに設定
		SAVE_DAYS=30						→　バックアップのDMP保存時間(単位:日)
